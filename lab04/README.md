    # Декоратор для кэширования результатов функции
def dec(func):
    cache = {}  # Словарь для хранения закэшированных результатов
    
    def wrapper(*args, **kwargs):
        # Формируем ключ для кэша на основе переданных аргументов
        # args - позиционные аргументы, kwargs - именованные
        key = (args, tuple(kwargs.items()))  # ключик для кэширки
        
        # Проверяем, есть ли уже результат для таких аргументов в кэше
        if key in cache:
            print(f"берем из кэша{args}")
            return cache[key]  # Возвращаем закэшированный результат
        
        # Если результата нет в кэше, вычисляем его
        print(f"считаем для аргументов{args}")
        result = func(*args, **kwargs)  # Вызываем исходную функцию
        cache[key] = result  # Сохраняем результат в кэш
        return result
    
    return wrapper


@dec  # Применяем декоратор кэширования к функции fib
def fib():
    # Функция, которая реализует замыкание для генерации последовательности Фибоначчи.
    a, b = 0, 1  # Начальные значения последовательности Фибоначчи
    
    def next():
       # Внутренняя функция (замыкание), которая при каждом вызове возвращает следующее число Фибоначчи.
        nonlocal a, b  # Указываем, что a и b принадлежат внешней функции fib
        res = a  # Запоминаем текущее значение для возврата
        a, b = b, a + b  # Вычисляем следующие значения последовательности
        return res  # Возвращаем текущее число Фибоначчи
    
    return next  # Возвращаем внутреннюю функцию (замыкание)


    # Создаем переменную с генератором последовательности Фибоначчи
fi = fib()

    # Выводим первые 7 чисел последовательности Фибоначчи
print(fi())  # 0 - первое число
print(fi())  # 1 - второе число
print(fi())  # 1 - третье число
print(fi())  # 2 - четвертое число
print(fi())  # 3 - пятое число
print(fi())  # 5 - шестое число
print(fi())  # 8 - седьмое число

#### Задание: создать декоратор для кэширования результатов выполнения функций.
```python
def opc(u=True):
    """
    декоратор с параметром
    u (bool): флаг включения/выключения кэширования
    True - кэширование включено
    False - кэширование отключено
    
    Возвращает:
        dec (function): декоратор, который кэширует результаты функции
    """
    def dec(func):
        cache = {}  # Словарь для хранения закэшированных результатов
        
        def wrapper(*args, **kwargs):
            """
            Обертка, реализующая кэширование результатов.
            Если кэширование отключено (u=False), просто вызываем функцию
            Если кэширование включено (u=True), проверяем наличие результата в кэше
            """
            # Если кэширование отключено, просто вызываем функцию без кэширования
            if not u:
                return func(*args, **kwargs)
            
            # Формируем ключ для кэша 
            # args - позиционные аргументы, kwargs - именованные
            key = (args, tuple(kwargs.items()))  # ключик для кэширки
            
            # Проверяем, есть ли результат в кэше
            if key in cache:
                print(f"берем из кэша{args}")  # Сообщение о взятии из кэша
                return cache[key]  # Возвращаем закэшированный результат
            
            # Если результата нет в кэше, вычисляем его
            print(f"считаем для аргументов{args}")  # Сообщение о вычислении
            result = func(*args, **kwargs)  # Вызываем исходную функцию
            cache[key] = result  # Сохраняем результат в кэш
            return result
        
        return wrapper
    return dec


# Декорируем с включенным кэшированием
@opc(True)  # Кэширование ВКЛЮЧЕНО
def fib(n):
    """
    Рекурсивная функция для вычисления числа Фибоначчи.
    """
    if n < 2:  # Базовый случай: fib(0)=0, fib(1)=1
        return n
    # Рекурсивный случай: fib(n) = fib(n-1) + fib(n-2)
    return fib(n-1) + fib(n-2)


# Декорируем с выключенным кэшированием
@opc(False)  # Кэширование ВЫКЛЮЧЕНО
def fib2(n):
    """
    Рекурсивная функция для вычисления числа Фибоначчи.
    """
    if n < 2:
        return n
    return fib2(n-1) + fib2(n-2)


# Декорируем без явного указания параметра
@opc
def fib3(n):
    """
    Рекурсивная функция для вычисления числа Фибоначчи.
    """
    if n < 2:
        return n
    return fib3(n-1) + fib3(n-2)


print(fib(3))   # Вычисление с кэшированием
print(fib2(3))  # Вычисление без кэширования
print(fib3(3))  # Будет ошибка или неожиданное поведение
```
