#### Задание: замыкание реализующее последовательность Фибоначчи.
```python 
    # Декоратор для кэширования результатов функции
def dec(func):
    cache = {}  # Словарь для хранения закэшированных результатов
    
    def wrapper(*args, **kwargs):
        # Формируем ключ для кэша на основе переданных аргументов
        # args - позиционные аргументы, kwargs - именованные
        key = (args, tuple(kwargs.items()))  # ключик для кэширки
        
        # Проверяем, есть ли уже результат для таких аргументов в кэше
        if key in cache:
            print(f"берем из кэша{args}")
            return cache[key]  # Возвращаем закэшированный результат
        
        # Если результата нет в кэше, вычисляем его
        print(f"считаем для аргументов{args}")
        result = func(*args, **kwargs)  # Вызываем исходную функцию
        cache[key] = result  # Сохраняем результат в кэш
        return result
    
    return wrapper


@dec  # Применяем декоратор кэширования к функции fib
def fib():
    # Функция, которая реализует замыкание для генерации последовательности Фибоначчи.
    a, b = 0, 1  # Начальные значения последовательности Фибоначчи
    
    def next():
       # Внутренняя функция (замыкание), которая при каждом вызове возвращает следующее число Фибоначчи.
        nonlocal a, b  # Указываем, что a и b принадлежат внешней функции fib
        res = a  # Запоминаем текущее значение для возврата
        a, b = b, a + b  # Вычисляем следующие значения последовательности
        return res  # Возвращаем текущее число Фибоначчи
    
    return next  # Возвращаем внутреннюю функцию (замыкание)


    # Создаем переменную с генератором последовательности Фибоначчи
fi = fib()

    # Выводим первые 7 чисел последовательности Фибоначчи
print(fi())  # 0 - первое число
print(fi())  # 1 - второе число
print(fi())  # 1 - третье число
print(fi())  # 2 - четвертое число
print(fi())  # 3 - пятое число
print(fi())  # 5 - шестое число
print(fi())  # 8 - седьмое число
```
